#include <stdio.h>

int main() {
    int year;

    // Prompt user for input
    printf("Enter a year: ");

    // Validate input
    if (scanf("%d", &year) != 1) {
        printf("Invalid input. Please enter a valid integer year.\n");
        return 1; // Exit with error
    }

    // Leap year check
    if (year % 400 == 0) {
        printf("%d is a leap year.\n", year);
    } 
    else if (year % 100 == 0) {
        printf("%d is not a leap year.\n", year);
    } 
    else if (year % 4 == 0) {
        printf("%d is a leap year.\n", year);
    } 
    else {
        printf("%d is not a leap year.\n", year);
    }

    return 0; // Successful execution
}
How it works
Divisible by 400 → Leap year (e.g., 2000, 2400).
Divisible by 100 but not 400 → Not a leap year (e.g., 1900, 2100).
Divisible by 4 but not 100 → Leap year (e.g., 2024, 2028).
Otherwise → Not a leap year.
